pipeline {
    agent any
    
    stages {
        stage('Checkout') {
            steps {
                checkout([$class: 'GitSCM', 
                          branches: [[name: '*/main']],
                          doGenerateSubmoduleConfigurations: false, 
                          extensions: [], 
                          submoduleCfg: [], 
                          userRemoteConfigs: [[url: 'https://github.com/sirisharapuru/Nodejs-app.git']]
                        ])
            }
        }
        
        stage('Build Docker image') {
            steps {
                sh "docker build -t image ."
            }
        }
        
        stage('Run container') {
            steps {
                sh "docker run -d -p 3000:3000 --name name image"
            }
        }
    }
