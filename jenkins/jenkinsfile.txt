pipeline {
    agent any
    stages {
        stage ("git"){
            steps{
                git credentialsId: 'git_credentials', url: 'https://github.com/sirisharapuru/Nodejs-app.git'
            }
        }
        stage('Build') {
            steps {
                checkout scm
                sh 'mvn clean install' //build using Maven
                archiveArtifacts artifacts: '**/target/*.jar', fingerprint: true //archive JAR files
            }
       }
    }
}
